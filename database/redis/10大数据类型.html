参考 ​

官网命令中文命令

String ​

sh
# 不存在时创建,存在时修改
set k1 v1

# 仅当k1不存在时创建
set k1 v1 nx

#仅当k1存在时修改
set k1 v1 xx

# 设置过期时间(秒|毫秒)
set k1 v1 (ex|px) 10

# 设置过期时间/unix时间戳
set k1 v1 exat 1700563342

#二次修改保留原来超时时间状态
set k1 v1 keepttl

#一次设置多个 不支持设置时间
mset k1 v1 k2 v2 ...

# 修改 k过期时间(毫秒|秒)
(pexpire|expire) k 10

#一次查询多个
mget k1 k2...

#取k1对应v下标 2 - 4(从0开始,前闭后闭)
getrange k1 2 4

#k1对应v 从下标2开始 替换为msg(长度不变)
setrange k1 2 <msg>

#get length of k
strlen k

#尾部追加
append k1 <xxx>

#查询旧的并修改
getset k1 v1
set k1 v1 get

如果是数字类型

# 自加1
incr k1

#加 n
incrby k1 <n>

# 减一
decr k1

#减 n
decrby k1 <n>

List ​

双端链表/左头右尾

#左端有序插入
lpush list1 1 2 3 4 5
#右端有序插入
rpush list1 1 2 3 4 5

#从左到右遍历list1 下边 n 到 m (0 ,-1 为全部)
lrange list1 n m

# 左端弹出一个值
lpop list1

# 右端弹出一个值
rpop list1

#获取list1中下标为index的值
lindex list1 <index>

# get size
llen list1

#删除list1中前n个值为val的元素(左边开始删)
lrem list1 <n> <val>

# 保留list1中下标n到m的元素,其他删除
ltrim list1 <n> <m>

#list1尾部弹出一个放进list2, 只能从前一个尾部 -> 后一个头部
rpoplpush list1 list2

# list1下标n设为val
lset list1 <n> <val>

#在list1 第一个val前面插一个newval
#返回插入后的size
linsert list1 before <val> <newval>

Hash ​

#键为k1,值为{id:1,name:jack}
hset k1 id 1 name jack

#查询 k1 的id属性
hget k1 id

#获取k1所有值
hgetall k1

#删除k1 id属性
hdel k1 id

#get size of k1
hlen k1

# 查询k1中是否有某一个属性
# return (1|0)
hexists k1 <filed>

# get keys of h1
hkeys k1

# get vals of h1
hvals k1

#must not exist then set
hsetnx k1 <filed> <val>

如果hash中某个属性的值是数字类型

# k1中id属性值加n
# return 加后的值
hincrby k1 id <n>

#加小数
hincrbyfloat k1 id <float>

Set ​

#set1中添加值
sadd set1 <v1> <v2> ...

# get all
smembers set1

# remove val
srem set1 <val>

# whether exist
sismember set1 <val>

#set1 中随机取n个数据
srandmember set1 <n>

#随机从set1中删除n个
spop set1 <n>

#set1中val移动到set2
smove set1 set2 <val>

#set1 - set2 
sdiff set1 set2 ...

# 并集
sunion set1 set2 ...

# 交集
sinter set1 set2 ...

# n个set交集的size
# limit限制size 最大数,不写为无限制
sintercard <n> (<set_key>){<n>} (limit <l>)?

Zset ​

有序set,每一个元素都对应一个分数

#add zset and insert
zadd <zset_k> (<scope_k> <k>)+

# 遍历下标从m -> n (0 -> -1 is all)
# withscores 同时查询分数
zrange <zset_k> <m> <n> (withscores)?

#反向遍历/查询(分数从大到小)
zrevrange <zset_k> 0 -1

# get where score between 40 and 70
# 将查询出的结果从offset开始查询n个(类似分页)
zrangebyscore <zset_k> 40 70 (withscores)? (limit <offset> <n>)?

#查询val的分数
zscore <zset_k> <val>

# size
zcard <zset_k>

# get count where score between min and max
zcount <zset_k> <min> <max>

# remove val
zrem <zset_k> <val>

# val的分数加n
zincrby <zset_k> <n> <val>

# get index of val
zrank <zset_k> <val>

#从n个set中的第一个非空set中弹出count个(最小|最大)元素
zmpop <n> (<zset_k>){<n>} (min|max) count <count>

# 弹出分数最大的count(default 1)个
zpopmax <zset_k> <count>?

# 弹出分数最小的count(default 1)个
zpopmax <zset_k> <count>?

# val 的反向下标(size - index)
zrevrank <zset_k> <val>

Stream ​

redis的消息队列

队列 ​

# 插入一条数据, (自动生成|指定id)
# id 由时间戳-序号俩部分组成 eg 1719602008673-0
# 一般自动生成id,因为新插入的数据id必须最大
xadd <key> (*|<id>) (<k> <v>)...

# 查询id指定范围( - 第一个 ; + 最后一个)
# count限制查询数量
xrange <key> (<min>|-) (<max>|+) (count <c>)?

# 反向(desc)
xrevrange <key> + -

# 删除
xdel <key> <id>

# length
xlen <key>

# 按照不同策略裁剪流, = 表示精确裁剪(default);~ 大概裁剪(少裁),不精准,快;
# 只有使用 ~ 大概裁剪时可以指定limit最大裁剪个数
# maxlen 裁剪后长度为n, = 必须为n, ~可以大于n
# minid 裁剪id小于n的元素
xtrim <key> (maxlen|minid) (=|~)? <n> (limit <c>)?

#从许多steam中查询消息(不删)
#block指定等待时间(毫秒) 0永久阻塞 如果没有可用消息,阻塞等待
# 只获取从指定id开始的数据 $是指只接受从现在开始的消息
xread [count <count>]? [block <b>] streams <key>+ (<id>|$)

消费者 ​

同一个消费组里的消费者不能读取同一条消息

#为qkey队列新建消费组gkey  $为只接受新消息  0为接受所有消息
xgroup create <qkey> <gkey> [$|0]

#gkey组中的消费者consumer从qkey中读取
xreadgroup group <gkey> <consumer> streams <qkey> > [count <c>]?

bitmap(位图) ​

二进制数组 (10010010101)(本质是string,可以进行string操作)

#第n位设为(1|0)
setbit <bit_k> <n> (1|0)

#  get 第n位
getbit <bit_k> <n>

# get length (字节)
strlen <bit_k>

# 1的数量
bitcount <bit_k>

#k1,k2中都为1的取出到新的newk
bitop and <newk> <k1> <k2>

HyperLogLog(基数统计) ​

去重数据集(有误差)

只记录个数不保存数据

#添加
pfadd <key> <val>...

#基数数量
pfcount <key>

#将俩个基数整合
pfmerge <newkey> <key1> <key2>

GEO(地理空间) ​

zset类型

#添加
geoadd <key> [经度 纬度 val]...

#返回指定val经纬度
geopos <key> <val>...

#指定val经纬度经过geohash算法后的值
geohash <key> <val>...

#返回指定俩个位置距离
geodist <key> <val1> <val2> [m|km|ft|mi]#ft英尺 mi英里

#返回指定经纬度n个单位半径内的位置
#withdist 同时返回距离 
#withcoord 返回每一个位置经纬度
#withhash 返回内部算法计算的52位值
#count 限制个数m
georadius <key> <经> <纬> <n> [m|km|ft|mi] [withdist] [withcoord] [withhash] [count <m>] [desc]

#同上,根据指定经纬度变为根据key中指定位置
georadiusbymember <key> <val> <n> [m|km|ft|mi] [withdist] [withcoord] [withhash] [count <m>] [desc]

bitfields(位域) ​

一般用不上

ascii表

#先定义一个String
set <key> <val>

#返回指定key中的offset偏移量 每8位一组  i有符号  u无符号
bitfield <key> get [i|u]8 <offset>

#指定key中的offset偏移量后面设置为val(二进制) 每8位一组 
bitfield <key> set i8 <offset> <val>
